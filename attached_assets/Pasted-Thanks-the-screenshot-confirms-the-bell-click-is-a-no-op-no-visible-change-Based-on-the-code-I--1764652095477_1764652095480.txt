Suggested change to InventoryTab.jsx (define handlers once and pass to all CardGroup instances)
```javascript name=src/components/InventoryTab.jsx url=https://github.com/BigDeckClub/BigDeckAppV3/blob/6ef33bc6ae7ea92c676d81be0e75cc5a533c0c55/src/components/InventoryTab.jsx
// near the top of the component where onLoadInventory is available:
const toggleAlertHandler = async (itemId) => {
  try {
    await fetch(`/api/inventory/${itemId}/toggle-alert`, { method: 'POST' });
    onLoadInventory?.();
  } catch (err) {
    console.error('Error toggling alert:', err);
  }
};

const setThresholdHandler = async (itemId, threshold) => {
  try {
    await fetch(`/api/inventory/${itemId}/set-threshold`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ threshold })
    });
    onLoadInventory?.();
  } catch (err) {
    console.error('Error setting threshold:', err);
  }
};

// Then, wherever you render <CardGroup ... />, ensure you pass:
<CardGroup
  ...
  onToggleLowInventory={toggleAlertHandler}
  onSetThreshold={setThresholdHandler}
/>
```
