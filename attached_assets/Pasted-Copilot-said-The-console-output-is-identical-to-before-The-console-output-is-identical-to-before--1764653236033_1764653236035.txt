app.post('/api/inventory/:id/toggle-alert', async (req, res) => {
  try {
    const { id } = req.params;
    
    // Get current value
    const before = await db.get('SELECT id, name, low_inventory_alert FROM inventory WHERE id = ? ', [id]);
    console.log('TOGGLE BEFORE:', before);
    
    // Toggle the value
    await db.run(
      'UPDATE inventory SET low_inventory_alert = NOT low_inventory_alert WHERE id = ?',
      [id]
    );
    
    // Get updated value
    const after = await db.get('SELECT * FROM inventory WHERE id = ?', [id]);
    console.log('TOGGLE AFTER:', after. low_inventory_alert);
    
    res.json(after);
  } catch (error) {
    console.error('TOGGLE ERROR:', error);
    res.status(500).json({ error: 'Failed to toggle alert' });
  }
});
